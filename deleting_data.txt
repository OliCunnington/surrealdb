-- Delete sub_category for the entire table

-- REMOVE FIELD -> only works with SCHEMA definitions
-- schemaless tables use SET ... NONE or UNSET
-- NULL means field is empty, NONE means field is non-existant

-- Delete sub_category for the entire table
UPDATE product MERGE {sub_category: NONE};
UPDATE product PATCH [
  {
    op: "remove",
    path: "sub_category"
  }
];

-- UNSET does not work in CONTENT, MERGE and PATCH clauses
-- need to use NONE (or REMOVE operation for PATCH)
